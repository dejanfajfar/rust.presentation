              macro_rules! MakeAnimal {
                ($name:ident, $body:expr) => {
                    struct $name {
                        name: String,
                        sound: String,
                    }
            
                    impl Animal for $name {
                        fn make_sound(&self) {
                            println!($body, self.sound);
                        }
                    }
                };
              }

              MakeAnimal!(Dog, "The dog says {}");
              MakeAnimal!(Cat, "The cat says {}");
            
              fn main() {
